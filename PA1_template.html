Use mean for that 5-minute interval to fit in all of the missing values in the dataset. Create a function name fill_na, which simpply accept 2 variables: the value of the step, and interval, if the step is NA, return the average of steps for that 5-minute interval, else return the value of the step itself.

fill_na <- function(step, interval){
  step_mean = average_steps_interval[(average_steps_interval$interval == interval),"mean"]
  if(is.na(step)){
    step_mean
  }else{
    step 
  }
}

fill_na_array <- function(steps, intervals){
  number = 0
  new_steps = steps
  for(step in steps){
    number = number +1
    new_steps[number] <- fill_na(steps[number], intervals[number])
  }
  new_steps
}

    day ="weekday"
